   and prop.priorLosses < 2000
   and prop.age <15

	prop.propertyRisk = "Low"

   and prop.priorLosses >= 2000
   and prop.age > 5  and prop.age < 15

	prop.propertyRisk = "Medium"

rule "PR6 Bricks"
ifrule prop.construction = "Brick"
   and prop.priorLosses >= 2000
   and prop.age > 15
   and prop.value > 100000

	prop.propertyRisk = "High"
